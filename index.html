DROP TABLE IF EXISTS species_list CASCADE;
CREATE TABLE species_list (id SERIAL PRIMARY KEY, botanical_name TEXT NOT NULL, chinese_name TEXT NOT NULL DEFAULT '', full_name TEXT NOT NULL UNIQUE, created_at TIMESTAMPTZ DEFAULT NOW());
